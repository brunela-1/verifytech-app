            const inactiveClasses = ['text-on-surface-variant', 'hover:bg-surface-container-high/50', 'hover:text-on-surface'];

            function switchTab(activeTab, inactiveTab) {
                // Remove inactive, add active
                inactiveClasses.forEach(cls => activeTab.classList.remove(...cls.split(' ')));
                activeClasses.forEach(cls => activeTab.classList.add(...cls.split(' ')));
                activeTab.setAttribute('aria-selected', 'true');
                
                // Adjust icon style
                const activeIcon = activeTab.querySelector('.material-symbols-outlined');
                if(activeIcon) activeIcon.style.fontVariationSettings = "'FILL' 1";

                // Remove active, add inactive
                activeClasses.forEach(cls => inactiveTab.classList.remove(...cls.split(' ')));
                inactiveClasses.forEach(cls => inactiveTab.classList.add(...cls.split(' ')));
                inactiveTab.setAttribute('aria-selected', 'false');
                
                // Adjust icon style
                const inactiveIcon = inactiveTab.querySelector('.material-symbols-outlined');
                if(inactiveIcon) inactiveIcon.style.fontVariationSettings = "'FILL' 0";
            }

            tabClient.addEventListener('click', () => switchTab(tabClient, tabTech));
            tabTech.addEventListener('click', () => switchTab(tabTech, tabClient));

        // Simple script to handle visual feedback on file upload area
        const fileUploadArea = document.querySelector('.group.border-dashed');
        const fileInput = document.getElementById('file-upload');

        ['dragenter', 'dragover', 'dragleave', 'drop'].forEach(eventName => {
            fileUploadArea.addEventListener(eventName, preventDefaults, false);
        });

        function preventDefaults (e) {
            e.preventDefault();
            e.stopPropagation();
        }

        ['dragenter', 'dragover'].forEach(eventName => {
            fileUploadArea.addEventListener(eventName, highlight, false);
        });

        ['dragleave', 'drop'].forEach(eventName => {
            fileUploadArea.addEventListener(eventName, unhighlight, false);
        });

        function highlight(e) {
            fileUploadArea.classList.add('bg-surface-container-low', 'border-primary');
        }

        function unhighlight(e) {
            fileUploadArea.classList.remove('bg-surface-container-low', 'border-primary');
        }

        fileUploadArea.addEventListener('drop', handleDrop, false);

        function handleDrop(e) {
            let dt = e.dataTransfer;
            let files = dt.files;
            handleFiles(files);
        }

        fileInput.addEventListener('change', function() {
            handleFiles(this.files);
        });

        function handleFiles(files) {
            if(files.length > 0) {
                 // Mocking file selection feedback
                 const textContainer = fileUploadArea.querySelector('.text-center');
                 textContainer.innerHTML = `
                    <span class="material-symbols-outlined text-secondary text-4xl mb-xs" style="font-variation-settings: 'FILL' 1;">check_circle</span>
                    <p class="font-label-md text-label-md text-on-surface mt-2">${files[0].name}</p>
                    <p class="font-label-sm text-label-sm text-primary mt-xs cursor-pointer hover:underline" onclick="resetFile(event)">Cambiar foto</p>
                 `;
            }
        }

<script>
        let isEditing = false;

        function toggleEditMode() {
            isEditing = !isEditing;
            const container = document.getElementById('profileFormContainer');
            const inputs = document.querySelectorAll('.editable-field');
            const actionBtns = document.getElementById('actionButtons');
            const toggleBtnText = document.getElementById('editBtnText');
            const toggleBtnIcon = document.querySelector('#editToggleBtn .material-symbols-outlined');

            if (isEditing) {
                container.classList.add('is-editing');
                inputs.forEach(input => input.removeAttribute('readonly'));
                inputs[0].focus(); // Focus first input
                
                actionBtns.classList.remove('hidden');
                // Small delay to allow display:block to apply before animating opacity
                setTimeout(() => actionBtns.classList.remove('opacity-0'), 10);
                
                toggleBtnText.textContent = "Cancel Edit";
                toggleBtnIcon.textContent = "close";
            } else {
                container.classList.remove('is-editing');
                inputs.forEach(input => input.setAttribute('readonly', 'true'));
                
                actionBtns.classList.add('opacity-0');
                setTimeout(() => actionBtns.classList.add('hidden'), 300); // Wait for transition
                
                toggleBtnText.textContent = "Editar Perfil";
                toggleBtnIcon.textContent = "edit";
                
                // Here you would normally reset values to original if cancelled
            }
        }

        function saveChanges() {
            // Placeholder for save logic
            const btn = document.querySelector('#actionButtons button.bg-primary');
            const originalText = btn.innerHTML;
            
            btn.innerHTML = `<span class="material-symbols-outlined animate-spin text-[18px]">progress_activity</span> Guardando...`;
            
            // Simulate API call
            setTimeout(() => {
                btn.innerHTML = originalText;
                toggleEditMode(); // Exit edit mode on save
                // Show a toast or notification in a real app
            }, 1000);
        }
    </script>
